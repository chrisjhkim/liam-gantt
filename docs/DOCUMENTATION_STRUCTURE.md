
### Git Hook 권장사항
```bash
# pre-commit hook
#!/bin/bash
oversized_files=$(find docs/ -name "*.md" -exec wc -l {} + | awk '$1 > 2000 {print $2}')
if [ ! -z "$oversized_files" ]; then
    echo "🚨 Error: Following files exceed 2000 lines:"
    echo "$oversized_files"
    echo "Please split these files before committing."
    exit 1
fi
```
